document		: NAME block

block			: OBRACE list CBRACE

list			: NAME COLON primitive list
                        | NAME COLON primitive
			| NAME block

primitive		: STRING
			| INT
			| REAL
			| BOOL
			| BINARY
			| LBRACKET array RBRACKET

array			: primitive
			| primitive COMMA array

Example:
Employee {
  2130498 {
     Name {
          First: "John"
          Middle: "Jacob"
          Last: "Doe"
          Modifiers {
               Prefix: "Doctor"
               Suffix: "Junior"
          }
     }
     Address {
        Number: 5684
        Street: "Main St."
        City: "San Francisco"
        State: "CA"
        ZIP: "94117"
     }
  }
}
